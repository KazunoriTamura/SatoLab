1function [ outputs ] = Topology_radar_mul20170510( sradar10, params)
  2%Topology_radar トポロジー法
  3% params はパラメータを格納した構造体。 以下の12個が必要。
  4% params.ncoh, params.Tr, params.cTr, params.HBR1, params.HBR2 
  5% params.Thcor, params.Thweight, params.Tc, params.NforF, params.CN 
  6% params.filt_length_low, params.filt_length_high
  7% radarはデータのアンラップ済み位相を格納したベクトル。(1xdatalength)
  8
  9%% ハイパス
 10% filt_length_high = (params.filt_length_high);
 11% % % % % % filt_high = [0; hanning(int8(params.filt_length_high)); 0];
 12% % % % % % filt_high = filt_high/sum(filt_high);
 13% % % % % % % filt_length_high = params.filt_length_high+2;
 14% filt_length_high=length(filt.filt_high);
 15% radar_filt_high = filter(filt.filt_high, 1, radar);
 16% 
 17% dradar=radar_filt_high(round(filt_length_high/2):end);
 18% radar_sel=radar(1:length(dradar));
 19% radar_wot=radar_sel-dradar;
 20% % mradar=max(radar);
 21% % % % % % % figure(1)
 22% % % % % % % plot((1:length(radar))/params.cTr,radar/mradar,'-x',(1:length(dradar))/params.cTr,dradar/mradar,'-o')
 23% % % % % % % xlabel('sample number')
 24% % % % % % % legend('処理前', 'トレンド')
 25% % % % % % % ylim([-1 1])
 26% % % % % % % drawnow
 27% %% ローパス
 28% % filt_length_low = (params.filt_length_low);
 29% % filt_low = hanning(filt_length_low);
 30% % filt_low = [0; hanning(filt_length_low); 0];
 31% % filt_low = filt_low/sum(filt_low);
 32% filt_length_low = length(filt.filt_low);
 33% outputs.t_delay_filt=round(filt_length_low/2)*params.cTr;
 34% % filt_low = filt_low/sum(filt_low);
 35% 
 36% sradar10 = filter(filt.filt_low, 1, radar_wot);
 37% figure(10)
 38% plot((1:length(radar_wot))/params.cTr,radar_wot,(1:length(sradar10))/params.cTr, sradar10)
 39% xlabel('sample number')
 40% legend('トレンド除去後', 'トレンド除去平滑化後')
 41% 
 42% % % % % ylim([-1 1])
 43% % % % % drawnow
 44M=size(sradar10,2);
 45times=(0:M-1)*params.cTr;
 46%% Differentiating radar signal
 47dsradar=zeros(size(sradar10));
 48dsradar(1:end-1) = diff(sradar10)/params.cTr;
 49
 50%% Detecting features
 51sradar = sradar10;
 52K=1;
 53K0=1;
 54K1=1;
 55K2=1;
 56K3=1;
 57%% For 1st diff
 58dif1_p=abs(sradar(2:M-1))>0.0;
 59dif2_1p=(sradar(2:M-1)-sradar(1:M-2))>0; 
 60dif2_1m=(sradar(2:M-1)-sradar(1:M-2))<0; 
 61dif2_2p=(sradar(3:M)-sradar(2:M-1))>0; 
 62dif2_2m=(sradar(3:M)-sradar(2:M-1))<0; 
 63
 64forc1p=ones(1,M);
 65forc1m=ones(1,M)*-1;
 66forci1p=ones(1,M)*1i;
 67forci1m=ones(1,M)*-1i;
 68forci12p=ones(1,M)*1i/2;
 69forci12m=ones(1,M)*-1i/2;
 70%% PK
 71ipeak0=find(dif1_p.*dif2_1p.*dif2_2m)+1;
 72tpeak0=times(ipeak0);
 73apeak0=sradar(ipeak0);
 74cpeak0=forc1m(ipeak0);
 75%% VL
 76ipeak10=find(dif1_p.*dif2_1m.*dif2_2p)+1;
 77tpeak10=times(ipeak10);
 78apeak10=sradar(ipeak10);
 79cpeak10=forc1p(ipeak10);
 80%% For 2nd diff
 81ds1_p=abs(dsradar(2:M-1))>0.0;
 82dsdif_1p= (dsradar(2:M-1)-dsradar(1:M-2))>0;
 83dsdif_1m= (dsradar(2:M-1)-dsradar(1:M-2))<0;
 84dsdif_2p=(dsradar(3:M)-dsradar(2:M-1))>0;
 85dsdif_2m=(dsradar(3:M)-dsradar(2:M-1))<0;
 86%% Rising/Inflection
 87ipeak20=find(ds1_p.*dsdif_1p.*dsdif_2m)+1;
 88tpeak20=(times(ipeak20)+times(ipeak20+1))/2;
 89apeak20=(sradar(ipeak20)+sradar(ipeak20+1))/2;
 90pkk=dsradar(2:M-1)>0;
 91ipkk=find(ds1_p.*dsdif_1p.*dsdif_2m.*pkk)+1;
 92forcp20=forci12m;
 93forcp20(ipkk)=forci1p(ipkk);
 94cpeak20=forcp20(ipeak20);
 95ipeak20=ipeak20+0.5;
 96%% Falling/Inflection
 97ipeak30=find(ds1_p.*dsdif_1m.*dsdif_2p)+1;
 98tpeak30=(times(ipeak30)+times(ipeak30+1))/2;
 99apeak30=(sradar(ipeak30)+sradar(ipeak30+1))/2;
100pkk=dsradar(2:M-1)<0;
101ipkk=find(ds1_p.*dsdif_1m.*dsdif_2p.*pkk)+1;
102forcp30=forci12p;
103forcp30(ipkk)=forci1m(ipkk);
104cpeak30=forcp30(ipeak30);
105ipeak30=ipeak30+0.5;
106tpeak000=[tpeak0 tpeak10 tpeak20 tpeak30];
107ipeak000=[ipeak0 ipeak10 ipeak20 ipeak30];
108apeak000=[apeak0 apeak10 apeak20 apeak30];
109cpeak000=[cpeak0 cpeak10 cpeak20 cpeak30];
110alls=sortrows([tpeak000' ipeak000' transpose(apeak000) transpose(cpeak000)]);
111tpeak=alls(:,1)';
112ipeak=alls(:,2)';
113apeak=transpose(alls(:,3));
114cpeak=transpose(alls(:,4));
115
116
117%%
118PEAKN = size(tpeak,2); %Number of feature points
119PEAKN0 = size(tpeak0,2); %Number of PK
120CORN = round(PEAKN0/times(end));
121acor=zeros(PEAKN,CORN);
122ibi = zeros(1,M);
123ibim = zeros(1,M);
124ibi0 = zeros(1,size(tpeak,2));
125
126%% Feature sequence 1400 is determined by 1.8s/1.285ms
127% 特徴点を中心に1400点の信号を抽出
128% 特徴点が700点目以前にある場合は、それ以前は0としている。
129gnas = zeros(PEAKN,params.NforF);
130
131for I=1:PEAKN
132    for J = -params.NforF/2:params.NforF/2-1
133        if round(ipeak(I))+J >= 1 && round(ipeak(I))+J <= M
134            gnas(I,J+params.NforF/2+1) = sradar(round(ipeak(I))+J);
135        end
136    end
137end
138
139%% checking topology
140
141cv = complex(zeros(PEAKN,params.CN)); % complex vector
142pcv = 0;
143ncv = 0;
144for I=1:PEAKN % the number of peaks
145    cv1 = complex(zeros(1,params.CN));
146    for K=-6:6 % uses 12 feature points around
147        if I+K >= 1 && I+K <= PEAKN % the feature point must be within a valid range 例えば、一つ目の特徴点の場合、K>=0以降しか計算しない
148            if abs(ipeak(I+K)-ipeak(I)) <= (params.CN-1)/2 % vector is 251 dimensional
149                if cpeak(I+K) == 1i/2 % change the sign of imagenary part % Fig. 8 to Fig. 9
150                    localcpeak = -1i/2;
151                else 
152                    if cpeak(I+K)== -1i/2 % change the sign of imagenary part % Fig. 8 to Fig. 9
153                        localcpeak = 1i/2;
154                    else
155                        localcpeak = cpeak(I+K);
156                    end
157                end
158                cv1(round(ipeak(I+K)-ipeak(I))+(params.CN-1)/2+1) = localcpeak;
159            end
160        end
161    end
162    cv(I,:) = cv1; % store the vector to matrix cv
163    inds=find(cv1~=0);
164    if(~isempty(inds))
165        cv_sel=[0, cv1(inds), 0];
166        inds_sel=[min(inds)-1, inds, max(inds)+1];
167        cv2_temp=interp1(inds_sel,cv_sel,min(inds)-1:1:max(inds)+1);
168        cv(I,min(inds):max(inds))=cv2_temp(2:end-1);
169    end    
170%     %  interpolation of phase
171%     for L=2:params.CN-1
172%         if cv1(L) == 0
173%             L0=1;
174%             while cv1(L-L0)==0 && L-L0 > 1
175%                 L0 = L0 + 1;
176%             end
177%             PL = L-L0;
178%             if cv1(PL)~=0
179%                 pcv = cv1(PL);
180%             else
181%                 pcv = 0;
182%             end
183%             
184%             L0=1;
185%             while cv1(L+L0)==0 && L+L0 < params.CN
186%                 L0 = L0 + 1;
187%             end
188%             NL = L+L0;
189%             if cv1(NL)~=0
190%                 ncv = cv1(NL);
191%             else
192%                 ncv = 0;
193%             end
194%             
195%             if pcv ~= 0 && ncv ~= 0
196%                 cv(I,L) = ((pcv)*(NL-L) + (ncv)*(L-PL))/(NL-PL);
197%             end
198%         end
199%     end
200end
201cv_temp=cv;
202cv = exp(1i*angle(cv));
203
204%%
205weightpeak = complex(zeros(PEAKN,PEAKN));
206for I=1:PEAKN
207    for J=1:PEAKN
208        if cpeak(I) == cpeak(J)
209            weightpeak(I,J) = cv(I,:)*cv(J,:)';
210        end
211    end
212end
213
214%%
215SEARCHN = round(PEAKN/times(end))*2; %Calcurate length for correlation 2秒の中に何個特徴点があるか？
216correlation = zeros(PEAKN,SEARCHN);
217maxcor = zeros(1,PEAKN);
218ibi3 = complex(zeros(1,PEAKN));
219chosenpair = zeros(1,PEAKN);
220avemaxcor = 1;
221for I=1:PEAKN-SEARCHN
222    for J=1:SEARCHN
223        vec1 = gnas(I,:);
224        vec2 = gnas(I+J,:);
225        vec1 = vec1 - real(mean(vec1));
226        vec2 = vec2 - real(mean(vec2));
227        correlation(I,J) = vec1*vec2'/sqrt(vec1*vec1')/sqrt(vec2*vec2');%Correlation between 特徴点Iに対する信号,特徴点I+Jに対する信号
228        if tpeak(I+J)-tpeak(I) < params.HBR1 || tpeak(I+J)-tpeak(I) > params.HBR2 || cpeak(I)~=cpeak(I+J) %Threshold for correlation
229            correlation(I,J) = 0;
230        end
231    end
232    
233    % weighting
234    mcv = 1:SEARCHN;
235    if I<11
236        weightcor = zeros(1,SEARCHN)+1;
237    else
238        bias0 = 0;
239        weightsum = 0;
240        for J=I-1:-1:I-10
241            if maxcor(J)>0
242                bias0 = bias0 + maxcor(J)*exp((J-I)/3); %忘却係数 3
243                weightsum = weightsum + exp((J-1)/3);
244            end
245        end
246        if weightsum ~= 0
247            bias0 = bias0 / weightsum;
248            weightcor = exp(-(mcv-bias0).^2/2/20^2); % 120 大きくなれば過去の影響を受けにくくなる。
249        else
250            weightcor = zeros(1,SEARCHN)+1;
251        end
252    end
253    [maxval,maxcor(I)] = max(correlation(I,:).*weightcor);
254    if max(correlation(I,:)) == 0
255        maxcor(I) = 0;
256    end
257    if maxval < params.Thcor
258        maxcor(I) = 0;
259    end
260    
261    ibi3(I) = tpeak(maxcor(I)+I)-tpeak(I);
262    chosenpair(I) = maxcor(I)+I; %Select the Feature point that has highest correlation
263end
264
265%%
266localweight = zeros(1,PEAKN);
267for I=1:PEAKN
268    if chosenpair(I) ~= 0
269        localweight(I) = max(0,real(weightpeak(I,chosenpair(I))));
270    else
271        localweight(I) = 0;
272    end
273end
274localweight = localweight/max(abs(localweight));
275
276ibi11 = complex(zeros(1,PEAKN));
277for I=1:PEAKN
278    if localweight(I) > params.Thweight
279        ibi11(I) = ibi3(I);
280    else
281        ibi11(I) = 0;
282    end
283end
284outputs.ibi11=ibi11;
285outputs.ipeak=ipeak;
286end
